\begin{activity}[]\label{act-binomformula}
\hypertarget{p-700}{}%
Give a combinatorial proof for the identity \(P(n,k) = \binom{n}{k}\cdot k!\).%
\par
\hypertarget{p-701}{}%
Conclude that%
\begin{equation*}
\binom{n}{k} = \frac{n!}{(n-k)!\cdot k!}\text{.}
\end{equation*}

\end{activity}
