\begin{activity}[]\label{simplifysumoffractions}
\hypertarget{p-1420}{}%
Express \(\frac{1}{x-3} + \frac{2}{x-2}\) as a single fraction.%
\par\smallskip%
\noindent\textbf{Solution.}\hypertarget{solution-200}{}\quad%
\hypertarget{p-1421}{}%
\(\frac{x-2 +2(x-3)}{(x-3)(x-2)}=\frac{3x-8}{x^2-5x+6}\)%
\end{activity}
